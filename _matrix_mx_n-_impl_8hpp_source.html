412 template <typename T>
413 void MatrixMxN<T>::Invert()
414 {
415  assert(IsSquare());
416 
417  // Computes inverse matrix using Gaussian elimination method.
418  // https://martin-thoma.com/solving-linear-equations-with-gaussian-elimination/
419  const size_t n = Rows();
420  MatrixMxN& a = *this;
421  MatrixMxN rhs = MakeIdentity(n);
422 
423  for (size_t i = 0; i < n; ++i)
424  {
425  // Search for maximum in this column
426  T maxEl = std::fabs(a(i, i));
427  size_t maxRow = i;
428 
429  for (size_t k = i + 1; k < n; ++k)
430  {
431  if (std::fabs(a(k, i)) > maxEl)
432  {
433  maxEl = std::fabs(a(k, i));
434  maxRow = k;
435  }
436  }
437 
438  // Swap maximum row with current row (column by column)
439  if (maxRow != i)
440  {
441  for (size_t k = i; k < n; ++k)
442  {
443  std::swap(a(maxRow, k), a(i, k));
444  std::swap(rhs(maxRow, k), rhs(i, k));
445  }
446  }
447 
448  // Make all rows except this one 0 in current column
449  for (size_t k = 0; k < n; ++k)
450  {
451  if (k == i)
452  {
453  continue;
454  }
455 
456  T c = -a(k, i) / a(i, i);
457 
458  for (size_t j = 0; j < n; ++j)
459  {
460  rhs(k, j) += c * rhs(i, j);
461 
462  if (i == j)
463  {
464  a(k, j) = 0;
465  }
466  else if (i < j)
467  {
468  a(k, j) += c * a(i, j);
469  }
470  }
471  }
472 
473  // Scale
474  for (size_t k = 0; k < n; ++k)
475  {
476  T c = 1 / a(k, k);
477 
478  for (size_t j = 0; j < n; ++j)
479  {
480  a(k, j) *= c;
481  rhs(k, j) *= c;
482  }
483  }
484  }
485 
486  Set(rhs);
487 }

618  T result = 1;
619 
620  for (size_t i = 0; i < n; ++i)
