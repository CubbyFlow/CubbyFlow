98 template <typename T>
99 void Array<T, 2>::Resize(const Size2& size, const T& initVal)
100 {
101  Array grid;
102  grid.m_data.resize(size.x * size.y, initVal);
103  grid.m_size = size;
104 
105  const size_t iMin = std::min(size.x, m_size.x);
106  const size_t jMin = std::min(size.y, m_size.y);
107  for (size_t j = 0; j < jMin; ++j)
108  {
109  for (size_t i = 0; i < iMin; ++i)
110  {
111  grid(i, j) = At(i, j);
112  }
113  }
114 
115  Swap(grid);
116 }
