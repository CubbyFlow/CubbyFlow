101 template <typename T, size_t N>
102 bool BoundingBox<T, N>::Intersects(const RayType& ray) const
103 {
104  T min = 0;
105  T max = std::numeric_limits<T>::max();
106  const VectorType& rayInvDir = T(1) / ray.direction;
107 
108  for (size_t i = 0; i < N; ++i)
109  {
110  T near = (lowerCorner[i] - ray.origin[i]) * rayInvDir[i];
111  T far = (upperCorner[i] - ray.origin[i]) * rayInvDir[i];
112 
113  if (near > far)
114  {
115  std::swap(near, far);
116  }
117 
118  min = near > min ? near : min;
119  max = far < max ? far : max;
120 
121  if (min > max)
122  {
123  return false;
124  }
125  }
126 
127  return true;
