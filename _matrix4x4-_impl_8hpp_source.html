147  const std::initializer_list<std::initializer_list<U>>& list)
148 {
149  const size_t height = list.size();
150  const size_t width = (height > 0) ? list.begin()->size() : 0;
151 
152  assert(width == 4);
153  assert(height == 4);
154 
155  auto rowIter = list.begin();
156  for (size_t i = 0; i < height; ++i)
157  {
158  assert(width == rowIter->size());
159 
160  auto colIter = rowIter->begin();
161  for (size_t j = 0; j < width; ++j)
162  {
163  (*this)(i, j) = static_cast<T>(*colIter);
164  ++colIter;
165  }
166 
167  ++rowIter;
168  }
169 }
