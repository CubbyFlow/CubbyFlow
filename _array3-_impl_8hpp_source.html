68 void Array<T, 3>::Set(const std::initializer_list<
69  std::initializer_list<std::initializer_list<T>>>& list)
70 {
71  const size_t depth = list.size();
72  const size_t height = (depth > 0) ? list.begin()->size() : 0;
73  const size_t width = (height > 0) ? list.begin()->begin()->size() : 0;
74 
75  Resize(Size3(width, height, depth));
76 
77  auto depthIter = list.begin();
78  for (size_t k = 0; k < depth; ++k)
79  {
80  assert(height == depthIter->size());
81 
82  auto heightIter = depthIter->begin();
83  for (size_t j = 0; j < height; ++j)
84  {
85  assert(width == heightIter->size());
86 
87  auto widthIter = heightIter->begin();
88  for (size_t i = 0; i < width; ++i)
89  {
90  (*this)(i, j, k) = *widthIter;
91  ++widthIter;
92  }
93  ++heightIter;
94  }
95  ++depthIter;
96  }

106 template <typename T>
107 void Array<T, 3>::Resize(const Size3& size, const T& initVal)
108 {
109  Array grid;
110  grid.m_data.resize(size.x * size.y * size.z, initVal);
111  grid.m_size = size;
112 
113  const size_t iMin = std::min(size.x, m_size.x);
114  const size_t jMin = std::min(size.y, m_size.y);
115  const size_t kMin = std::min(size.z, m_size.z);
116 
117  for (size_t k = 0; k < kMin; ++k)
118  {
119  for (size_t j = 0; j < jMin; ++j)
120  {
121  for (size_t i = 0; i < iMin; ++i)
122  {
123  grid(i, j, k) = At(i, j, k);
124  }
125  }
126  }
127 
128  Swap(grid);
129 }
