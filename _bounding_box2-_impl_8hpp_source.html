118 {
119  T min = 0;
120  T max = std::numeric_limits<T>::max();

122  const Vector2<T>& rayInvDir = ray.direction.RDiv(1);
123 
124  for (size_t i = 0; i < 2; ++i)
125  {
126  T near = (lowerCorner[i] - ray.origin[i]) * rayInvDir[i];
127  T far = (upperCorner[i] - ray.origin[i]) * rayInvDir[i];

135  max = std::min(far, max);
