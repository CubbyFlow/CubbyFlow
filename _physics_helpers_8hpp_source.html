19 inline Vector2D ComputeDragForce(double dragCoefficient, double radius,
20  const Vector2D& velocity)
21 {
22  // Stoke's drag force assuming our Reynolds number is very low.
23  // http://en.wikipedia.org/wiki/Drag_(physics)#Very_low_Reynolds_numbers:_Stokes.27_drag
24  return -6.0 * PI_DOUBLE * dragCoefficient * radius * velocity;
25 }

35 template <size_t N>
36 Vector<double, N> ProjectAndApplyFriction(const Vector<double, N>& vel,
37  const Vector<double, N>& normal,
38  double frictionCoefficient)
39 {
40  Vector<double, N> velt = vel.Projected(normal);
41  if (velt.LengthSquared() > 0)
42  {
43  const double veln = std::max(-vel.Dot(normal), 0.0);
44  velt *= std::max(1.0 - frictionCoefficient * veln / velt.Length(), 0.0);
45  }
46 
47  return velt;
48 }

50 inline double ComputePressureFromEos(double density, double targetDensity,
51  double eosScale, double eosExponent,
52  double negativePressureScale)
53 {
54  // See Murnaghan-Tait equation of state from
55  // https://en.wikipedia.org/wiki/Tait_equation
56  double p = eosScale / eosExponent *
57  (std::pow((density / targetDensity), eosExponent) - 1.0);
58 
59  // Negative pressure scaling
60  if (p < 0)
61  {
62  p *= negativePressureScale;
63  }
64 
65  return p;
66 }
