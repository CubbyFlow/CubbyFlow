74 T NearestArraySampler<T, R, 2>::operator()(const Vector2<R>& pt) const
75 {
76  ssize_t i, j;
77  R fx, fy;
78 
79  assert(m_gridSpacing.x > std::numeric_limits<R>::epsilon());
80  assert(m_gridSpacing.y > std::numeric_limits<R>::epsilon());
81 
82  const Vector2<R> normalizedX = (pt - m_origin) / m_gridSpacing;
83 
84  const ssize_t iSize = static_cast<ssize_t>(m_accessor.size().x);
85  const ssize_t jSize = static_cast<ssize_t>(m_accessor.size().y);
86 
87  GetBarycentric(normalizedX.x, 0, iSize - 1, &i, &fx);
88  GetBarycentric(normalizedX.y, 0, jSize - 1, &j, &fy);
89 
90  i = std::min(static_cast<ssize_t>(i + fx + 0.5), iSize - 1);
91  j = std::min(static_cast<ssize_t>(j + fy + 0.5), jSize - 1);
92 
93  return m_accessor(i, j);
94 }

338 template <typename T, typename R>
339 T CubicArraySampler<T, R, 2>::operator()(const Vector2<R>& pt) const
340 {
341  ssize_t i, j;
342  R fx, fy;
343 
344  assert(m_gridSpacing.x > std::numeric_limits<R>::epsilon());
345  assert(m_gridSpacing.y > std::numeric_limits<R>::epsilon());
346 
347  const Vector2<R> normalizedX = (pt - m_origin) / m_gridSpacing;
348 
349  const ssize_t iSize = static_cast<ssize_t>(m_accessor.size().x);
350  const ssize_t jSize = static_cast<ssize_t>(m_accessor.size().y);
351 
352  GetBarycentric(normalizedX.x, 0, iSize - 1, &i, &fx);
353  GetBarycentric(normalizedX.y, 0, jSize - 1, &j, &fy);
354 
355  const ssize_t is[4] = { std::max(i - 1, ZERO_SSIZE), i,
356  std::min(i + 1, iSize - 1),
357  std::min(i + 2, iSize - 1) };
358  const ssize_t js[4] = { std::max(j - 1, ZERO_SSIZE), j,
359  std::min(j + 1, jSize - 1),
360  std::min(j + 2, jSize - 1) };
361 
362  // Calculate in i direction first
363  T values[4];
364 
365  for (int n = 0; n < 4; ++n)
366  {
367  values[n] = MonotonicCatmullRom(
368  m_accessor(is[0], js[n]), m_accessor(is[1], js[n]),
369  m_accessor(is[2], js[n]), m_accessor(is[3], js[n]), fx);
370  }
371 
372  return MonotonicCatmullRom(values[0], values[1], values[2], values[3], fy);
373 }
