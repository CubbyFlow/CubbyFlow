579  Matrix<T, Rows, Cols> a{ m.GetDerived() };
580

582 
583  for (size_t i = 0; i < m.GetRows(); ++i)
584  {
585  // Search for maximum in this column
586  T maxEl = std::fabs(a(i, i));
587  size_t maxRow = i;
588 
589  for (size_t k = i + 1; k < m.GetRows(); ++k)
590  {

592  {
593  maxEl = std::fabs(a(k, i));
594  maxRow = k;
595  }
596  }
597 
598  // Swap maximum row with current row (column by column)
599  if (maxRow != i)
600  {
601  for (size_t k = i; k < m.GetRows(); ++k)
602  {
603  std::swap(a(maxRow, k), a(i, k));
604  }
605  result *= -1;
606  }
607 
608  // Make all rows below this one 0 in current column
609  for (size_t k = i + 1; k < m.GetRows(); ++k)
610  {
611  T c = -a(k, i) / a(i, i);
612

614  {
615  if (i == j)
616  {
617  a(k, j) = 0;
618  }
619  else

621  a(k, j) += c * a(i, j);
622  }
623  }
624  }
625  }
626

628  {

630  }
631

800 template <typename T, size_t Rows, size_t Cols, typename Derived>
801 template <typename M>
802 void MatrixExpression<T, Rows, Cols, Derived>::Inverse(
803  const MatrixExpression<T, Rows, Cols, Derived>& m,
804  std::enable_if_t<
805  (Rows > 4 && Cols > 4) || IsMatrixSizeDynamic<Rows, Cols>(), M>& result)
806 {
807  // Computes inverse matrix using Gaussian elimination method.
808  // https://martin-thoma.com/solving-linear-equations-with-gaussian-elimination/
809  Matrix<T, Rows, Cols> a{ m.GetDerived() };
810 
811  using ConstType = MatrixConstant<T, Rows, Cols>;
812 
813  result = MatrixDiagonal<T, Rows, Cols, ConstType>{ ConstType{
814  a.GetRows(), a.GetCols(), 1 } };
815  const size_t n = m.GetRows();
816 
817  for (size_t i = 0; i < n; ++i)
818  {
819  // Search for maximum in this column
820  T maxEl = std::fabs(a(i, i));
821  size_t maxRow = i;
822 
823  for (size_t k = i + 1; k < n; ++k)
824  {
825  if (std::fabs(a(k, i)) > maxEl)
826  {
827  maxEl = std::fabs(a(k, i));
828  maxRow = k;
829  }
830  }
831 
832  // Swap maximum row with current row (column by column)
833  if (maxRow != i)
834  {
835  for (size_t k = i; k < n; ++k)
836  {
837  std::swap(a(maxRow, k), a(i, k));
838  }
839 
840  for (size_t k = 0; k < n; ++k)
841  {
842  std::swap(result(maxRow, k), result(i, k));
843  }
844  }
845 
846  // Make all rows except this one 0 in current column
847  for (size_t k = 0; k < n; ++k)
848  {
849  if (k == i)
850  {
851  continue;
852  }
853 
854  T c = -a(k, i) / a(i, i);
855 
856  for (size_t j = 0; j < n; ++j)
857  {
858  result(k, j) += c * result(i, j);
859 
860  if (i == j)
861  {
862  a(k, j) = 0;
863  }
864  else if (i < j)
865  {
866  a(k, j) += c * a(i, j);
867  }
868  }
869  }
870 
871  // Scale
872  for (size_t k = 0; k < n; ++k)
873  {
874  T c = 1 / a(k, k);
875 
876  for (size_t j = 0; j < n; ++j)
877  {
878  a(k, j) *= c;
879  result(k, j) *= c;
880  }
881  }
882  }
883 }
