50 template <typename T, typename VE>
51 T MatrixCSRVectorMul<T, VE>::operator[](size_t i) const
52 {
53  auto rp = m_m.RowPointersBegin();
54  auto ci = m_m.ColumnIndicesBegin();
55  auto nnz = m_m.NonZeroBegin();
56 
57  const size_t colBegin = rp[i];
58  const size_t colEnd = rp[i + 1];
59 
60  T sum = 0;
61  for (size_t jj = colBegin; jj < colEnd; ++jj)
62  {
63  size_t j = ci[jj];
64  sum += nnz[jj] * m_v[j];
65  }
66 
67  return sum;
68 }

213 template <typename T>
214 void MatrixCSR<T>::Compress(
215  const std::initializer_list<std::initializer_list<T>>& list, T epsilon)
216 {
217  const size_t numRows = list.size();
218  const size_t numCols = (numRows > 0) ? list.begin()->size() : 0;
219 
220  m_size = { numRows, numCols };
221  m_nonZeros.clear();
222  m_rowPointers.clear();
223  m_columnIndices.clear();
224 
225  auto rowIter = list.begin();
226  for (size_t i = 0; i < numRows; ++i)
227  {
228  assert(numCols == rowIter->size());
229  m_rowPointers.push_back(m_nonZeros.size());
230 
231  auto colIter = rowIter->begin();
232  for (size_t j = 0; j < numCols; ++j)
233  {
234  if (std::fabs(*colIter) > epsilon)
235  {
236  m_nonZeros.push_back(*colIter);
237  m_columnIndices.push_back(j);
238  }
239 
240  ++colIter;
241  }
242 
243  ++rowIter;
244  }
245 
246  m_rowPointers.push_back(m_nonZeros.size());
247 }

250 template <typename E>
251 void MatrixCSR<T>::Compress(const MatrixExpression<T, E>& other, T epsilon)
252 {
253  const size_t numRows = other.Rows();
254  const size_t numCols = other.Cols();
255 
256  m_size = { numRows, numCols };
257  m_nonZeros.clear();
258  m_columnIndices.clear();
259 
260  const E& expression = other();
261 
262  for (size_t i = 0; i < numRows; ++i)
263  {
264  m_rowPointers.push_back(m_nonZeros.size());
265 
266  for (size_t j = 0; j < numCols; ++j)
267  {
268  T val = expression(i, j);
269 
270  if (std::fabs(val) > epsilon)
271  {
272  m_nonZeros.push_back(val);
273  m_columnIndices.push_back(j);
274  }
275  }
276  }
277 
278  m_rowPointers.push_back(m_nonZeros.size());
279 }

287 template <typename T>
288 void MatrixCSR<T>::AddElement(const Element& element)
289 {
290  const ssize_t numRowsToAdd =
291  static_cast<ssize_t>(element.i) - static_cast<ssize_t>(m_size.x) + 1;
292  if (numRowsToAdd > 0)
293  {
294  for (ssize_t i = 0; i < numRowsToAdd; ++i)
295  {
296  AddRow({}, {});
297  }
298  }
299 
300  m_size.y = std::max(m_size.y, element.j + 1);
301 
302  const size_t rowBegin = m_rowPointers[element.i];
303  const size_t rowEnd = m_rowPointers[element.i + 1];
304 
305  auto colIdxIter =
306  std::lower_bound(m_columnIndices.begin() + rowBegin,
307  m_columnIndices.begin() + rowEnd, element.j);
308  auto offset = colIdxIter - m_columnIndices.begin();
309 
310  m_columnIndices.insert(colIdxIter, element.j);
311  m_nonZeros.insert(m_nonZeros.begin() + offset, element.value);
312 
313  for (size_t i = element.i + 1; i < m_rowPointers.size(); ++i)
314  {
315  ++m_rowPointers[i];
316  }
317 }

319 template <typename T>
320 void MatrixCSR<T>::AddRow(const NonZeroContainerType& nonZeros,
321  const IndexContainerType& columnIndices)
322 {
323  assert(nonZeros.size() == columnIndices.size());
324 
325  ++m_size.x;
326 
327  // TODO: Implement zip iterator
328  std::vector<std::pair<T, size_t>> zipped;
329  for (size_t i = 0; i < nonZeros.size(); ++i)
330  {
331  zipped.emplace_back(nonZeros[i], columnIndices[i]);
332  m_size.y = std::max(m_size.y, columnIndices[i] + 1);
333  }
334 
335  std::sort(zipped.begin(), zipped.end(),
336  [](std::pair<T, size_t> a, std::pair<T, size_t> b) {
337  return a.second < b.second;
338  });
339 
340  for (size_t i = 0; i < zipped.size(); ++i)
341  {
342  m_nonZeros.push_back(zipped[i].first);
343  m_columnIndices.push_back(zipped[i].second);
344  }
345 
346  m_rowPointers.push_back(m_nonZeros.size());
347 }

987 template <typename T>
988 size_t MatrixCSR<T>::HasElement(size_t i, size_t j) const
989 {
990  if (i >= m_size.x || j >= m_size.y)
991  {
992  return std::numeric_limits<size_t>::max();
993  }
994 
995  const size_t rowBegin = m_rowPointers[i];
996  const size_t rowEnd = m_rowPointers[i + 1];
997 
998  const auto iter = BinaryFind(m_columnIndices.begin() + rowBegin,
999  m_columnIndices.begin() + rowEnd, j);
1000  if (iter != m_columnIndices.begin() + rowEnd)
1001  {
1002  return static_cast<size_t>(iter - m_columnIndices.begin());
1003  }
1004 
1005  return std::numeric_limits<size_t>::max();

1008 template <typename T>
1009 template <typename Op>
1010 MatrixCSR<T> MatrixCSR<T>::BinaryOp(const MatrixCSR& m, Op op) const
1011 {
1012  assert(m_size == m.m_size);
1013 
1014  MatrixCSR ret;
1015 
1016  for (size_t i = 0; i < m_size.x; ++i)
1017  {
1018  std::vector<size_t> col;
1019  std::vector<double> nnz;
1020 
1021  auto colIterA = m_columnIndices.begin() + m_rowPointers[i];
1022  auto colIterB = m.m_columnIndices.begin() + m.m_rowPointers[i];
1023  auto colEndA = m_columnIndices.begin() + m_rowPointers[i + 1];
1024  auto colEndB = m.m_columnIndices.begin() + m.m_rowPointers[i + 1];
1025  auto nnzIterA = m_nonZeros.begin() + m_rowPointers[i];
1026  auto nnzIterB = m.m_nonZeros.begin() + m.m_rowPointers[i];
1027 
1028  while (colIterA != colEndA || colIterB != colEndB)
1029  {
1030  if (colIterB == colEndB || *colIterA < *colIterB)
1031  {
1032  col.push_back(*colIterA);
1033  nnz.push_back(op(*nnzIterA, 0));
1034  ++colIterA;
1035  ++nnzIterA;
1036  }
1037  else if (colIterA == colEndA || *colIterA > *colIterB)
1038  {
1039  col.push_back(*colIterB);
1040  nnz.push_back(op(0, *nnzIterB));
1041  ++colIterB;
1042  ++nnzIterB;
1043  }
1044  else
1045  {
1046  assert(*colIterA == *colIterB);
1047  col.push_back(*colIterB);
1048  nnz.push_back(op(*nnzIterA, *nnzIterB));
1049  ++colIterA;
1050  ++nnzIterA;
1051  ++colIterB;
1052  ++nnzIterB;
1053  }
1054  }
1055 
1056  ret.AddRow(nnz, col);
1057  }
1058 
1059  return ret;
1060 }
