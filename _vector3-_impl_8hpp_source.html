655 template <typename T>
656 Vector<T, 3> MonotonicCatmullRom(const Vector<T, 3>& v0, const Vector<T, 3>& v1,
657  const Vector<T, 3>& v2, const Vector<T, 3>& v3,
658  T f)
659 {
660  static const T two = static_cast<T>(2);
661  static const T three = static_cast<T>(3);
662 
663  Vector<T, 3> d1 = (v2 - v0) / two;
664  Vector<T, 3> d2 = (v3 - v1) / two;
665  Vector<T, 3> D1 = v2 - v1;
666 
667  if (std::fabs(D1.x) < std::numeric_limits<float>::epsilon() ||
668  Sign(D1.x) != Sign(d1.x) || Sign(D1.x) != Sign(d2.x))
669  {
670  d1.x = d2.x = 0;
671  }
672 
673  if (std::fabs(D1.y) < std::numeric_limits<float>::epsilon() ||
674  Sign(D1.y) != Sign(d1.y) || Sign(D1.y) != Sign(d2.y))
675  {
676  d1.y = d2.y = 0;
677  }
678 
679  if (std::fabs(D1.z) < std::numeric_limits<float>::epsilon() ||
680  Sign(D1.z) != Sign(d1.z) || Sign(D1.z) != Sign(d2.z))
681  {
682  d1.z = d2.z = 0;
683  }
684 
685  Vector<T, 3> a3 = d1 + d2 - two * D1;
686  Vector<T, 3> a2 = three * D1 - two * d1 - d2;
687  Vector<T, 3> a1 = d1;
688  Vector<T, 3> a0 = v1;
689 
690  return a3 * Cubic(f) + a2 * Square(f) + a1 * f + a0;
691 }
