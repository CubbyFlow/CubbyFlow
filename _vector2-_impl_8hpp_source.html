603 template <typename T>
604 Vector<T, 2> MonotonicCatmullRom(const Vector<T, 2>& v0, const Vector<T, 2>& v1,
605  const Vector<T, 2>& v2, const Vector<T, 2>& v3,
606  T f)
607 {
608  static const T two = static_cast<T>(2);
609  static const T three = static_cast<T>(3);
610 
611  Vector<T, 2> d1 = (v2 - v0) / two;
612  Vector<T, 2> d2 = (v3 - v1) / two;
613  Vector<T, 2> D1 = v2 - v1;
614 
615  if (std::fabs(D1.x) < std::numeric_limits<T>::epsilon() ||
616  Sign(D1.x) != Sign(d1.x) || Sign(D1.x) != Sign(d2.x))
617  {
618  d1.x = d2.x = 0;
619  }
620 
621  if (std::fabs(D1.y) < std::numeric_limits<T>::epsilon() ||
622  Sign(D1.y) != Sign(d1.y) || Sign(D1.y) != Sign(d2.y))
623  {
624  d1.y = d2.y = 0;
625  }
626 
627  Vector<T, 2> a3 = d1 + d2 - two * D1;
628  Vector<T, 2> a2 = three * D1 - two * d1 - d2;
629  Vector<T, 2> a1 = d1;
630  Vector<T, 2> a0 = v1;
631 
632  return a3 * Cubic(f) + a2 * Square(f) + a1 * f + a0;
633 }
