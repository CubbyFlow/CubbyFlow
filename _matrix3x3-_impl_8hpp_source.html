95 void Matrix<T, 3, 3>::Set(
96  const std::initializer_list<std::initializer_list<U>>& list)
97 {
98  const size_t height = list.size();
99  const size_t width = (height > 0) ? list.begin()->size() : 0;
100 
101  assert(width == 3);
102  assert(height == 3);
103 
104  auto rowIter = list.begin();
105  for (size_t i = 0; i < height; ++i)
106  {
107  assert(width == rowIter->size());
108 
109  auto colIter = rowIter->begin();
110  for (size_t j = 0; j < width; ++j)
111  {
112  (*this)(i, j) = static_cast<T>(*colIter);
113  ++colIter;
114  }
115 
116  ++rowIter;
117  }
118 }

486 {

492  m.m_elements[1] =

494  m.m_elements[2] =

496  m.m_elements[3] =

498  m.m_elements[4] =

504  m.m_elements[7] =
