415 template <typename T, size_t M, size_t N>
416 void Matrix<T, M, N>::Invert()
417 {
418  assert(IsSquare());
419 
420  // Computes inverse matrix using Gaussian elimination method.
421  // https://martin-thoma.com/solving-linear-equations-with-gaussian-elimination/
422  const size_t n = Rows();
423  Matrix& a = *this;
424  Matrix rhs = MakeIdentity();
425 
426  for (size_t i = 0; i < n; ++i)
427  {
428  // Search for maximum in this column
429  T maxEl = std::fabs(a(i, i));
430  size_t maxRow = i;
431 
432  for (size_t k = i + 1; k < n; ++k)
433  {
434  if (std::fabs(a(k, i)) > maxEl)
435  {
436  maxEl = std::fabs(a(k, i));
437  maxRow = k;
438  }
439  }
440 
441  // Swap maximum row with current row (column by column)
442  if (maxRow != i)
443  {
444  for (size_t k = i; k < n; ++k)
445  {
446  std::swap(a(maxRow, k), a(i, k));
447  }
448 
449  for (size_t k = 0; k < n; ++k)
450  {
451  std::swap(rhs(maxRow, k), rhs(i, k));
452  }
453  }
454 
455  // Make all rows except this one 0 in current column
456  for (size_t k = 0; k < n; ++k)
457  {
458  if (k == i)
459  {
460  continue;
461  }
462 
463  T c = -a(k, i) / a(i, i);
464 
465  for (size_t j = 0; j < n; ++j)
466  {
467  rhs(k, j) += c * rhs(i, j);
468 
469  if (i == j)
470  {
471  a(k, j) = 0;
472  }
473  else if (i < j)
474  {
475  a(k, j) += c * a(i, j);
476  }
477  }
478  }
479 
480  // Scale
481  for (size_t k = 0; k < n; ++k)
482  {
483  T c = 1 / a(k, k);
484 
485  for (size_t j = 0; j < n; ++j)
486  {
487  a(k, j) *= c;
488  rhs(k, j) *= c;
489  }
490  }
491  }
492 
493  Set(rhs);
494 }

582 template <typename T, size_t M, size_t N>
583 T Matrix<T, M, N>::Determinant() const
584 {
585  assert(IsSquare());
586 
587  // Computes inverse matrix using Gaussian elimination method.
588  // https://martin-thoma.com/solving-linear-equations-with-gaussian-elimination/
589  const size_t n = Rows();
590  Matrix a(*this);
591  T result = 1;
592 
593  for (size_t i = 0; i < n; ++i)
594  {
595  // Search for maximum in this column
596  T maxEl = std::fabs(a(i, i));
597  size_t maxRow = i;
598 
599  for (size_t k = i + 1; k < n; ++k)
600  {
601  if (std::fabs(a(k, i)) > maxEl)
602  {
603  maxEl = std::fabs(a(k, i));
604  maxRow = k;
605  }
606  }
607 
608  // Swap maximum row with current row (column by column)
609  if (maxRow != i)
610  {
611  for (size_t k = i; k < n; ++k)
612  {
613  std::swap(a(maxRow, k), a(i, k));
614  }
615 
616  result *= -1;
617  }
618 
619  // Make all rows below this one 0 in current column
620  for (size_t k = i + 1; k < n; ++k)
621  {
622  T c = -a(k, i) / a(i, i);
623 
624  for (size_t j = i; j < n; ++j)
625  {
626  if (i == j)
627  {
628  a(k, j) = 0;
629  }
630  else
631  {
632  a(k, j) += c * a(i, j);
633  }
634  }
635  }
636  }
637 
638  for (size_t i = 0; i < n; ++i)
639  {
640  result *= a(i, i);
641  }
642 
643  return result;
644 }
