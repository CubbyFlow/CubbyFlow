101 struct GetCoordinatesAndWeights
102 {
103  using ScalarType = typename GetScalarType<T>::value;
104 
105  template <typename Coords, typename Weights, typename... RemainingIndices>
106  static void Call(Coords& c, Weights& w, Vector<size_t, N> i,
107  Vector<ScalarType, N> t, T acc, RemainingIndices... idx)
108  {
109  using Next = GetCoordinatesAndWeights<T, N, I - 1>;

112  Next::Call(c, w, i, t, acc * t[I - 1], 1, idx...);

117 struct GetCoordinatesAndWeights<T, N, 1>

120 
121  template <typename Coords, typename Weights, typename... RemainingIndices>

123  Vector<ScalarType, N> t, T acc, RemainingIndices... idx)

133 template <typename T, size_t N, size_t I>
134 struct GetCoordinatesAndGradientWeights
135 {
136  template <typename Coords, typename Weights, typename... RemainingIndices>
137  static void Call(Coords& c, Weights& w, Vector<size_t, N> i, Vector<T, N> t,
138  Vector<T, N> acc, RemainingIndices... idx)
139  {
140  Vector<T, N> w0 = Vector<T, N>::MakeConstant(1 - t[I - 1]);
141  w0[I - 1] = -1;
142  Vector<T, N> w1 = Vector<T, N>::MakeConstant(t[I - 1]);
143  w1[I - 1] = 1;
144 
145  using Next = GetCoordinatesAndGradientWeights<T, N, I - 1>;
146  Next::Call(c, w, i, t, ElemMul(acc, w0), 0, idx...);
147  Next::Call(c, w, i, t, ElemMul(acc, w1), 1, idx...);
148  }
149 };

234 template <typename T, size_t N>
235 typename NearestArraySampler<T, N>::CoordIndexType
236 NearestArraySampler<T, N>::GetCoordinate(const VectorType& pt) const
237 {
238  Vector<ssize_t, N> is;
239  VectorType npt = ElemMul(pt - m_gridOrigin, m_invGridSpacing);
240  Vector<ssize_t, N> size = m_view.Size().template CastTo<ssize_t>();
241 
242  for (size_t i = 0; i < N; ++i)
243  {
244  Vector<ScalarType, N> ts;
245  GetBarycentric(npt[i], 0, size[i], is[i], ts[i]);
246  is[i] =
247  std::min(static_cast<ssize_t>(is[i] + ts[i] + 0.5), size[i] - 1);
248  }
249 
250  return is.template CastTo<size_t>();
251 }
