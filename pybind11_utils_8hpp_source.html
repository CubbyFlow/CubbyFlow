618 inline void ParseGridResizeParams(pybind11::args args, pybind11::kwargs kwargs,
619  Size2& resolution, Vector2D& gridSpacing,
620  Vector2D& gridOrigin)
621 {
622  // See if we have list of parameters
623  if (args.size() <= 3)
624  {
625  if (args.size() > 0)
626  {
627  resolution = ObjectToSize2(pybind11::object(args[0]));
628  }
629  if (args.size() > 1)
630  {
631  gridSpacing = ObjectToVector2D(pybind11::object(args[1]));
632  }
633  if (args.size() > 2)
634  {
635  gridOrigin = ObjectToVector2D(pybind11::object(args[2]));
636  }
637  }
638  else
639  {
640  throw std::invalid_argument("Too many arguments.");
641  }
642 
643  // Parse out keyword args
644  if (kwargs.contains("resolution"))
645  {
646  resolution = ObjectToSize2(pybind11::object(kwargs["resolution"]));
647  }
648  if (kwargs.contains("gridSpacing"))
649  {
650  gridSpacing = ObjectToVector2D(pybind11::object(kwargs["gridSpacing"]));
651  }
652  if (kwargs.contains("gridOrigin"))
653  {
654  gridOrigin = ObjectToVector2D(pybind11::object(kwargs["gridOrigin"]));
655  }
656  if (kwargs.contains("domainSizeX"))
657  {
658  double domainSizeX = kwargs["domainSizeX"].cast<double>();
659  gridSpacing.Set(domainSizeX / static_cast<double>(resolution.x));
660  }
661 }
