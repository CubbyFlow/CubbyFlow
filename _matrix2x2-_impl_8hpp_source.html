82 template <typename T>
83 template <typename U>
84 void Matrix<T, 2, 2>::Set(
85  const std::initializer_list<std::initializer_list<U>>& list)
86 {
87  const size_t height = list.size();
88  const size_t width = (height > 0) ? list.begin()->size() : 0;
89 
90  assert(width == 2);
91  assert(height == 2);
92 
93  auto rowIter = list.begin();
94  for (size_t i = 0; i < height; ++i)
95  {
96  assert(width == rowIter->size());
97 
98  auto colIter = rowIter->begin();
99  for (size_t j = 0; j < width; ++j)
100  {
101  (*this)(i, j) = static_cast<T>(*colIter);
102  ++colIter;
103  }
104 
105  ++rowIter;
106  }
107 }
