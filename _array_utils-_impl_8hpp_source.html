85 template <typename T>
86 void ExtrapolateToRegion(const ConstArrayAccessor2<T>& input,
87  const ConstArrayAccessor2<char>& valid,
88  unsigned int numberOfIterations,
89  ArrayAccessor2<T> output)
90 {
91  const Size2 size = input.size();
92 
93  assert(size == valid.size());
94  assert(size == output.size());
95 
96  Array2<char> valid0(size);
97  Array2<char> valid1(size);
98 
99  valid0.ParallelForEachIndex([&](size_t i, size_t j) {
100  valid0(i, j) = valid(i, j);
101  output(i, j) = input(i, j);
102  });
103 
104  for (unsigned int iter = 0; iter < numberOfIterations; ++iter)
105  {
106  valid0.ForEachIndex([&](size_t i, size_t j) {
107  T sum = Zero<T>();
108  unsigned int count = 0;
109 
110  if (!valid0(i, j))
111  {
112  if (i + 1 < size.x && valid0(i + 1, j))
113  {
114  sum += output(i + 1, j);
115  ++count;
116  }
117 
118  if (i > 0 && valid0(i - 1, j))
119  {
120  sum += output(i - 1, j);
121  ++count;
122  }
123 
124  if (j + 1 < size.y && valid0(i, j + 1))
125  {
126  sum += output(i, j + 1);
127  ++count;
128  }
129 
130  if (j > 0 && valid0(i, j - 1))
131  {
132  sum += output(i, j - 1);
133  ++count;
134  }
135 
136  if (count > 0)
137  {
138  output(i, j) =
139  sum / static_cast<typename ScalarType<T>::value>(count);
140  valid1(i, j) = 1;
141  }
142  }
143  else
144  {
145  valid1(i, j) = 1;
146  }
147  });
148 
149  valid0.Swap(valid1);
150  }
