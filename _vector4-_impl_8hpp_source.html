667 template <typename T>
668 Vector<T, 4> MonotonicCatmullRom(const Vector<T, 4>& v0, const Vector<T, 4>& v1,
669  const Vector<T, 4>& v2, const Vector<T, 4>& v3,
670  T f)
671 {
672  static const T two = static_cast<T>(2);
673  static const T three = static_cast<T>(3);
674 
675  Vector<T, 4> d1 = (v2 - v0) / two;
676  Vector<T, 4> d2 = (v3 - v1) / two;
677  Vector<T, 4> D1 = v2 - v1;
678 
679  if (std::fabs(D1.x) < std::numeric_limits<float>::epsilon() ||
680  Sign(D1.x) != Sign(d1.x) || Sign(D1.x) != Sign(d2.x))
681  {
682  d1.x = d2.x = 0;
683  }
684 
685  if (std::fabs(D1.y) < std::numeric_limits<float>::epsilon() ||
686  Sign(D1.y) != Sign(d1.y) || Sign(D1.y) != Sign(d2.y))
687  {
688  d1.y = d2.y = 0;
689  }
690 
691  if (std::fabs(D1.z) < std::numeric_limits<float>::epsilon() ||
692  Sign(D1.z) != Sign(d1.z) || Sign(D1.z) != Sign(d2.z))
693  {
694  d1.z = d2.z = 0;
695  }
696 
697  if (std::fabs(D1.w) < std::numeric_limits<float>::epsilon() ||
698  Sign(D1.w) != Sign(d1.w) || Sign(D1.w) != Sign(d2.w))
699  {
700  d1.w = d2.w = 0;
701  }
702 
703  Vector<T, 4> a3 = d1 + d2 - two * D1;
704  Vector<T, 4> a2 = three * D1 - two * d1 - d2;
705  Vector<T, 4> a1 = d1;
706  Vector<T, 4> a0 = v1;
707 
708  return a3 * Cubic(f) + a2 * Square(f) + a1 * f + a0;
709 }
