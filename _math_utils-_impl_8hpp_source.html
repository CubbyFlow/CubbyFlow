154 {

201 template <typename S, typename T>
202 S CatmullRom(const S& f0, const S& f1, const S& f2, const S& f3, T t)
203 {
204  S d1 = (f2 - f0) / 2;
205  S d2 = (f3 - f1) / 2;
206  S D1 = f2 - f1;
207 
208  S a3 = d1 + d2 - 2 * D1;
209  S a2 = 3 * D1 - 2 * d1 - d2;
210  S a1 = d1;
211  S a0 = f1;
212 
213  return a3 * Cubic(t) + a2 * Square(t) + a1 * t + a0;
214 }

216 template <typename T>
217 T MonotonicCatmullRom(const T& f0, const T& f1, const T& f2, const T& f3, T t)
218 {
219  T d1 = (f2 - f0) / 2;
220  T d2 = (f3 - f1) / 2;
221  T D1 = f2 - f1;
222 
223  if (std::fabs(D1) < std::numeric_limits<double>::epsilon())
224  {
225  d1 = d2 = 0;
226  }
227 
228  if (Sign(D1) != Sign(d1))
229  {
230  d1 = 0;
231  }
232 
233  if (Sign(D1) != Sign(d2))
234  {
235  d2 = 0;
236  }
237 
238  T a3 = d1 + d2 - 2 * D1;
239  T a2 = 3 * D1 - 2 * d1 - d2;
240  T a1 = d1;
241  T a0 = f1;
242 
243  return a3 * Cubic(t) + a2 * Square(t) + a1 * t + a0;
